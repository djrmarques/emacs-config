* Hydra
  #+BEGIN_SRC emacs-lisp
      (use-package hydra
      :ensure t
      )


    (defun get-org-conf ()
      (interactive)
      (find-file org-config-file)
      )
      
    (defun get-init ()
      (interactive)
      (find-file init-file)
      )
      

    (defhydra hydra-files (global-map "<f2>")
      "Find Files"
      ("c" get-org-conf "Find org conf")
      ("i" get-init "Find org conf")
      ("k" kill-all-buffers "Kill al buffers")
)

  #+END_SRC
